795 });

799 argc = 0;

801 FOREACH_ARG_IN_VA_ARGS(ignore, const char*, arg, args,

829 exit(1);

834 if (execvp(argv[0], (char* const *)argv) < 0)

836 INTERNAL_ECHO(stderr, " -- "ERROR" Could not execute child process: %s\n", strerror(errno));
837 exit(1);
838 }
839 }

842 for (;;)

844 int status = 0;
845 wait(&status);

847 if (WIFEXITED(status))

872int _isCBuildModified(const char* sourcePath, const char* binaryPath)
873{
874#ifdef _WIN32
875 FILETIME path1_time, path2_time;

878 if (!GetFileTime(path1_fd, NULL, NULL, &path1_time))

881 exit(1);
882 }

884 fd_close(path1_fd);

895#else
896 struct stat statbuf;

899 {
