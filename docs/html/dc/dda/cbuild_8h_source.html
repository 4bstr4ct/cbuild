838void _cmd(int ignore, ...)
839{
840#ifdef _WIN32
841 assert(!"TODO: implement _cmd with Windows WIN32 API!");
842#else
843 unsigned long long argc = 0;
844 va_list args;
845
846 FOREACH_ARG_IN_VA_ARGS(ignore, const char*, arg, args,
847 {
848 ++argc;
849 });
850
851 assert(argc >= 1);
852 const char** argv = (const char**)malloc((argc + 1) * sizeof(const char*));
853 argc = 0;
854
855 FOREACH_ARG_IN_VA_ARGS(ignore, const char*, arg, args,
856 {
857 argv[argc++] = arg;
858 });
859
860 argv[argc] = NULL;
861 assert(argc >= 1);
862 _ProcessId childProcessId = fork();
863
864 if (childProcessId == -1)
865 {
866 INTERNAL_ECHO(stderr, " -- "CBUILD_ERROR" Could not fork child process: %s\n", strerror(errno));
867 exit(1);
868 }
869
870 if (childProcessId == 0)
871 {
872 if (execvp(argv[0], (char* const *)argv) < 0)
873 {
874 INTERNAL_ECHO(stderr, " -- "CBUILD_ERROR" Could not execute child process: %s\n", strerror(errno));
875 exit(1);
876 }
877 }
878 else
879 {
880 for (;;)
881 {
882 int status = 0;
883 wait(&status);
884
885 if (WIFEXITED(status))
886 {
887 const int exitStatus = WEXITSTATUS(status);
888
889 if (exitStatus != 0)
890 {
891 INTERNAL_ECHO(stderr, " -- "CBUILD_ERROR" Child process exited with %d code\n", exitStatus);
892 exit(1);
893 }
894
895 break;
896 }
897
898 if (WIFSIGNALED(status))
899 {
900 INTERNAL_ECHO(stderr, " -- "CBUILD_ERROR" Child process was terminated by %d signal\n", WTERMSIG(status));
901 exit(1);
902 }
903 }
904 }
905
906 free(argv);
907#endif
908}
