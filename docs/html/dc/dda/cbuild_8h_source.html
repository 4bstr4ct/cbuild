182const char* _join(const char* const separator, ...)
183{
184 const unsigned long long separatorLength = strlen(separator);
185 signed long long separatorsCount = -1;
186 unsigned long long length = 0;
187 va_list args;
188
189 FOREACH_ARG_IN_VA_ARGS(separator, const char*, arg, args,
190 {
191 length += strlen(arg);
192 ++separatorsCount;
193 });
194
195 char* buffer = (char*)malloc((length + separatorsCount * separatorLength + 1) * sizeof(char));
196 length = 0;
197
198 FOREACH_ARG_IN_VA_ARGS(separator, const char*, arg, args,
199 {
200 const unsigned long long argLength = strlen(arg);
201 memcpy(buffer + length, arg, argLength);
202 length += argLength;
203
204 if (separatorsCount > 0)
205 {
206 memcpy(buffer + length, separator, separatorLength);
207 length += separatorLength;
208 --separatorsCount;
209 }
210 });
211
212 buffer[length] = '\0';
213 return buffer;
214}

727void _cmd(int ignore, ...)
728{
729#ifdef _WIN32
730 assert(!"TODO: implement _cmd with Windows WIN32 API!");
731#else
732 unsigned long long argc = 0;
733 va_list args;
734
735 FOREACH_ARG_IN_VA_ARGS(ignore, const char*, arg, args,
736 {
737 ++argc;
738 });
739
740 assert(argc >= 1);
741 const char** argv = (const char**)malloc((argc + 1) * sizeof(const char*));
742 argc = 0;
743
744 FOREACH_ARG_IN_VA_ARGS(ignore, const char*, arg, args,
745 {
746 argv[argc++] = arg;
747 });
748
749 argv[argc] = NULL;
750 assert(argc >= 1);
751 pid_t childProcessId = fork();
752
753 if (childProcessId == -1)
754 {
755#ifndef CBUILD_NOECHO
756 ECHO(stderr, CBUILD_ERROR_LABEL" Failed to fork child process: "CBUILD_ERROR("%s")"\n", strerror(errno));
757#endif
758
759 free(argv);
760 exit(1);
761 }
762
763 if (childProcessId == 0)
764 {
765 if (execvp(argv[0], (char* const *)argv) < 0)
766 {
767#ifndef CBUILD_NOECHO
768 ECHO(stderr, CBUILD_ERROR_LABEL" Failed to execute child process: "CBUILD_ERROR("%s")"\n", strerror(errno));
769#endif
770
771 free(argv);
772 exit(1);
773 }
774 }
775 else
776 {
777 for (;;)
778 {
779 int status = 0;
780 wait(&status);
781
782 if (WIFEXITED(status))
783 {
784 const int exitStatus = WEXITSTATUS(status);
785
786 if (exitStatus != 0)
787 {
788#ifndef CBUILD_NOECHO
789 ECHO(stderr, CBUILD_ERROR_LABEL" Child process exited with code "CBUILD_ERROR("%d")"\n", exitStatus);
790#endif
791
792 free(argv);
793 exit(1);
794 }
795
796 break;
797 }
798
799 if (WIFSIGNALED(status))
800 {
801#ifndef CBUILD_NOECHO
802 ECHO(stderr, CBUILD_ERROR_LABEL" Child process was terminated by "CBUILD_ERROR("%d")" signal\n", WTERMSIG(status));
803#endif
804
805 free(argv);
806 exit(1);
807 }
808 }
809 }
810
811 free(argv);
812#endif
813}
